public class CombineManager : MonoBehaviour
{
    // Match detection algorithm
    public List<Item> FindMatches(Square square)
    {
        List<Item> horizontal = GetHorizontalMatches(square);
        List<Item> vertical = GetVerticalMatches(square);
        
        if (horizontal.Count >= 3 || vertical.Count >= 3)
        {
            return CombineMatches(horizontal, vertical);
        }
        return new List<Item>();
    }
    
    // Special item generation logic
    private Item GenerateSpecialItem(List<Item> matches)
    {
        if (matches.Count == 4) return CreateStripedItem();
        if (matches.Count == 5) return CreateBombItem();
        if (matches.Count >= 6) return CreateRainbowItem();
    }
}

Match Detection Engine

Optimized grid-based matching algorithm with support for:

  • Pattern Recognition: Horizontal, vertical, L-shape, T-shape combinations
  • Special Item Creation: Rule-based generation for 4-match, 5-match, and mega-matches
  • Chain Reactions: Recursive cascade system for combo chains
  • Performance: O(n) complexity with spatial hashing optimization

public class InitScript : MonoBehaviour
{
    // Singleton pattern
    public static InitScript Instance;
    
    // Core game state
    public GameState currentState;
    public int currentLevel;
    public int score;
    public int moves;
    
    // Initialization pipeline
    void Awake()
    {
        InitializeSingleton();
        LoadPlayerData();
        InitializeSubsystems();
        LoadLevel(currentLevel);
    }
    
    // Game state management
    public void SetGameState(GameState newState)
    {
        currentState = newState;
        OnGameStateChanged?.Invoke(newState);
    }
}
